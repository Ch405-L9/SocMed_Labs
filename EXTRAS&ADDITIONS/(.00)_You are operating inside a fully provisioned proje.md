```ts
export function resolveToolId(
  naturalLanguage: string
): { id: ToolId | null; confidence: number } {
  const text = naturalLanguage.toLowerCase();
  let best: { id: ToolId | null; confidence: number } = { id: null, confidence: 0 };
  for (const cfg of TOOL_ALIASES) {
    for (const alias of cfg.aliases) {
      if (text.includes(alias)) {
        if (1 > best.confidence) best = { id: cfg.id, confidence: 1 };
      } else {
        // fuzzy substring scoring; simple heuristic
        const words = alias.split(/\s+/);
        const matched = words.filter(w => text.includes(w)).length;
        const score = matched / words.length;
        if (score > best.confidence && score >= 0.5) {
          best = { id: cfg.id, confidence: score };
        }
      }
    }
  }
  return best;
}
```

### 2.3 Ambiguity handling and clarification

In `src/infra/router.ts`, implement orchestration that:[^1_8][^1_7]

- Resolves aliases with `resolveToolId`.
- If confidence is below a threshold (e.g., 0.7), returns a structured “needs clarification” error instead of executing anything.
- Enforces governance flags (unsafe, ambiguous) before invoking tools.

```ts
// src/infra/router.ts
import { resolveToolId, ToolId } from "./tool-aliases";
import { evaluateGovernance } from "./governance";

export interface RoutedToolCall {
  toolId: ToolId;
  input: unknown;
}

export interface RouterResult {
  needsClarification: boolean;
  clarificationPrompt?: string;
  routedCall?: RoutedToolCall;
}

export function routeUserIntent(
  userQuery: string,
  parsedInput: unknown
): RouterResult {
  const { id, confidence } = resolveToolId(userQuery);

  if (!id || confidence < 0.7) {
    return {
      needsClarification: true,
      clarificationPrompt:
        "Your request maps to multiple operations (web optimization, SEO, and social content). " +
        "Please specify whether you want: 'seo audit', 'web optimization', 'social content', or 'brand rules'."
    };
  }

  const gov = evaluateGovernance(userQuery);
  if (gov.requiresClarification) {
    return {
      needsClarification: true,
      clarificationPrompt: gov.message
    };
  }

  return {
    needsClarification: false,
    routedCall: {
      toolId: id,
      input: parsedInput
    }
  };
}
```

## 3. Governance model (medium-level control)

Leverage the BADGR brand rules and enforcement policy already encoded in `social-profile-tools.json` (e.g., `badgrbrandrules`, `ollamavalidate` failure modes and guardrails).[^1_4][^1_8][^1_6]

Create a governance module that:

- Flags obviously unsafe operations (e.g., direct system file manipulation, PII scraping).
- Flags ambiguous goals (“do my marketing” with no platform or offer).
- Surfaces an `auditTag` and `riskLevel` that can be logged by the MCP server.

```ts
// src/infra/governance.ts
export interface GovernanceAssessment {
  unsafe: boolean;
  requiresClarification: boolean;
  message?: string;
  auditTag: string;
  riskLevel: "low" | "medium" | "high";
}

const UNSAFE_PATTERNS = [
  "delete all",
  "wipe database",
  "scrape emails",
  "bypass security",
  "exploit"
];

export function evaluateGovernance(query: string): GovernanceAssessment {
  const lower = query.toLowerCase();
  const unsafeMatch = UNSAFE_PATTERNS.find(p => lower.includes(p));
  if (unsafeMatch) {
    return {
      unsafe: true,
      requiresClarification: true,
      message:
        "The requested operation appears unsafe (" +
        unsafeMatch +
        "). Please rephrase with a specific, non-destructive goal.",
      auditTag: "unsafe_pattern",
      riskLevel: "high"
    };
  }

  if (lower.includes("do my marketing") || lower.trim().length < 10) {
    return {
      unsafe: false,
      requiresClarification: true,
      message:
        "Your request is too vague for automated execution. " +
        "Please specify platform (e.g., TikTok, LinkedIn), target audience, and main offer.",
      auditTag: "ambiguous_goal",
      riskLevel: "medium"
    };
  }

  return {
    unsafe: false,
    requiresClarification: false,
    auditTag: "ok",
    riskLevel: "low"
  };
}
```

This matches the “medium governance” requirement and aligns with existing BADGR enforcement policies which forbid fabricated stats, emojis, etc.[^1_4][^1_6]

### 4.2 Chunking strategy (15% smaller than “standard”)

The instructions say to reduce chunk size 15% below norms.[^1_5]

If your typical chunk size is 1,000–1,200 tokens per chunk, target ~850–1,000 tokens:

- For conceptual knowledge: split by H2/H3 headings and cap each chunk at ~800–900 words.
- For tool schema docs: one tool per chunk to maintain schema integrity.
- For usage examples: one scenario per chunk (e.g., “seo-audit on homepage”).

Update your indexing script (Node or Python) to enforce a lower token/character threshold when concatenating paragraphs.[^1_5]

### 4.3 RAG index format

In line with `rag_setup_instructions_viral_videos.md`, store RAG indices as JSON with:[^1_5]

- `id`: unique string, like `"viral:chunks:platform:tiktok:001"`.
- `collection`: `"viral-videos"` or `"mcp-tools"`.
- `docType`: `"conceptual" | "tool-schema" | "usage-example"`.
- `toolId`: for tool-related docs; `null` for generic conceptual docs.
- `tags`: string array.
- `text`: the actual chunk.
- `vector`: numeric embedding array (from your local embedding model).

Example element for a `seo-audit` schema chunk:

```json
{
  "id": "mcp-tools:seo-audit:schema:001",
  "collection": "mcp-tools",
  "docType": "tool-schema",
  "toolId": "seo-audit",
  "tags": ["seo", "audit", "web", "kit", "schema"],
  "text": "Tool: seo-audit\n\nPurpose: Perform a structured SEO and technical audit for a single URL...\n\nInput schema:\n- url (string, required)\n- crawl_depth (integer, optional, default 1)...",
  "vector": [0.0123, -0.0045, 0.0789, ...]
}
```

```ts
  "stop losing",
  "introducing",
  "did you know"
];

const ALLOWED_DOLLARS = new Set(["$197", "$500"]);
const BIO_Q_EXCEPTIONS = new Set<Platform>(["tiktok"]);

export function runGuard(
  raw: string,
  platform: Platform,
  fieldType: FieldType
): GuardResult {
  const brandVoice = fieldType !== "post" || platform !== "generic";
  let content = raw.trim();
  const violations: string[] = [];
  const autoResolved: string[] = [];
  const scores: Record<string, number> = {};

  // G1 — emoji (auto-strip)
  const emojiRx = /[\u{1F000}-\u{1FFFF}]|[\u{2600}-\u{27BF}]|[\uFE00-\uFE0F]/gu;
  if (emojiRx.test(content)) {
    content = content.replace(emojiRx, "").replace(/[ \t]{2,}/g, " ").trim();
    autoResolved.push("emoji_stripped");
    scores["emoji"] = 70;
  } else {
    scores["emoji"] = 100;
  }

  let brandScore = 100;
  let voiceScore = 100;
  let statsScore = 100;
  let copyScore = 100;

  if (brandVoice) {
    // G2 — forbidden name forms
    let namePenalty = 0;
    for (const f of FORBIDDEN_NAMES) {
      if (content.includes(f)) {
        violations.push(`forbidden_name:${f.replace(/ /g, "_")}`);
        namePenalty += 20;
      }
    }
    brandScore = Math.max(0, 100 - namePenalty);
    scores["brand_name"] = brandScore;

    // G3 — question opener (exception: TikTok bio)
    const isBioField = ["bio", "about", "description", "headline"].includes(
      fieldType
    );
    const isBioQException =
      isBioField && BIO_Q_EXCEPTIONS.has(platform as Platform);
    const firstLine = content.split(/\n/)[^2_0].trim();
    if (!isBioQException && /^[^.!?]*\?/u.test(firstLine)) {
      violations.push("question_opener");
      voiceScore = 55;
    }
    scores["voice"] = voiceScore;

    // G5 — percentage stats
    const pcts = [...content.matchAll(/\d+\s*%/g)].map(m => m[^2_0]);
    if (pcts.length > 0) {
      violations.push(`fabricated_stat:${pcts.join(",")}`);
      statsScore = 40;
    }
    // G6 — dollar amounts
    const dollars = [...content.matchAll(/\$(\d[\d,]*)/g)].map(m => m[^2_0]);
    const badDollars = dollars.filter(d => !ALLOWED_DOLLARS.has(d));
    if (badDollars.length > 0) {
      violations.push(`suspicious_dollar:${badDollars.join(",")}`);
      statsScore = Math.min(statsScore, 40);
    }
    scores["stats"] = statsScore;

    // G7 — fluff
    const lower = content.toLowerCase();
    const fluffHit = FLUFF.filter(p => lower.includes(p));
    if (fluffHit.length > 0) {
      violations.push(`fluff:${fluffHit.join("|")}`);
      copyScore = Math.max(0, 100 - fluffHit.length * 10);
    }
    scores["copy"] = copyScore;
  } else {
    scores["brand_name"] = 100;
    scores["voice"] = 100;
    scores["stats"] = 100;
    scores["copy"] = 100;
  }

  // G4 — char limit (always enforced)
  const limits = CHAR_LIMITS[platform];
  if (limits) {
    const limit = limits[fieldType];
    if (limit !== undefined) {
      if (content.length > limit) {
        violations.push(`char_overflow:${content.length}>${limit}`);
        scores["char_limit"] = Math.max(
          0,
          Math.round((limit / content.length) * 100)
        );
      } else {
        scores["char_limit"] = 100;
      }
    } else {
      scores["char_limit"] = 100;
    }
  } else {
    scores["char_limit"] = 100;
  }

  const vals = Object.values(scores);
  const confidence =
    vals.length > 0
      ? vals.reduce((a, b) => a + b, 0) / vals.length / 100
      : 0.5;

  return {
    content,
    scores,
    violations,
    autoResolved,
    ready: violations.length === 0,
    confidence
  };
}
```

```
async function callOllama(
  ollama: Ollama,
  model: string,
  system: string,
  user: string,
  timeoutMs: number
): Promise<{ content: string; error?: string }> {
  try {
    const res = await withTimeout(
      ollama.chat({
        model,
        messages: [
          { role: "system", content: system },
          { role: "user", content: user }
        ],
        options: { temperature: 0.3 }
      }),
      timeoutMs
    );
    return { content: res.message.content };
  } catch (e) {
    return { content: "", error: e instanceof Error ? e.message : String(e) };
  }
}

export function classify(request: string): Classification {
  const lo = request.toLowerCase();

  const editW =
    (/\b(tweak|edit|fix|change|update|adjust|revise|modify|swap)\b/.test(lo)
      ? 3
      : 0) +
    (/\b(small|minor|quick|just)\b/.test(lo) ? 2 : 0);
  const genW =
    (/\b(generate|create|write|draft|new|make)\b/.test(lo) ? 3 : 0) +
    (/\b(bio|caption|hashtag|hook|cta|tagline)\b/.test(lo) ? 2 : 0);
  const setupW =
    (/\b(profile|setup|build|complete|full)\b/.test(lo) ? 3 : 0) +
    (/\b(name|handle|link|strategy)\b/.test(lo) ? 2 : 0);
  const assetW = /\b(image|photo|logo|banner|cover|avatar|resize|convert|imagemagick|px)\b/.test(
    lo
  )
    ? 3
    : 0;
  const anlsW = /\b(audit|analyz|diagnose|review|teardown|fix.?list|what.?wrong)\b/.test(
    lo
  )
    ? 3
    : 0;

  const scores: Record<TaskType, number> = {
    small_edit: editW,
    generation: genW,
    profile_setup: setupW,
    asset_pipeline: assetW,
    analysis: anlsW
  };

  let best: TaskType = "generation";
  let bestScore = -1;
  for (const [t, s] of Object.entries(scores) as [TaskType, number][]) {
    if (s > bestScore) {
      bestScore = s;
      best = t;
    }
  }

  const words = request.split(/\s+/).length;
  const platforms =
    lo.match(/tiktok|instagram|youtube|facebook|\bx\b|linkedin/g)?.length ?? 0;
  const complexity =
    words < 10 && platforms <= 1
      ? "low"
      : words > 30 || platforms > 1
      ? "high"
      : "medium";

  const RAG_TRIGGERS =
    /viral.?tech.?video|use.*video.?system|use.*video.?pattern|use.*video.?doc|video.?strategy|from.?knowledge.?base|use.?rag/i;
  const requires_rag = RAG_TRIGGERS.test(request);

  return { task_type: best, complexity, requires_rag, domain: "social" };
}

// ---- Execute core ----
export async function executePipeline(input: {
  ollama: Ollama;
  classification: Classification;
  content?: string;
  platform?: Platform;
  field_type?: FieldType;
  context?: string;
}): Promise<BadgrOutput> {
  const { ollama, classification, content, platform, context } = input;
  const tt = classification.task_type;
  const route = ROUTING[tt];
  const ft: FieldType =
    input.field_type ??
    (platform && PRIMARY_FIELD[platform]) ??
    "bio";

  let system = buildSystemPrompt(tt, platform, ft);

  if (classification.requires_rag) {
    try {
      const ragQuery = `viral tech video system context: ${
        content ?? ""
      } ${context ?? ""}`.trim();
      const chunks = await queryRag(ollama, ragQuery, 4);
      if (chunks.length > 0) {
        const ragBlock = chunks.map(c => c.text).join("\n\n---\n\n");
        system += `\n\nVIRAL TECH VIDEO KNOWLEDGE BASE (retrieved context):\n\n${ragBlock}`;
      }
    } catch {
      // proceed without RAG
    }
  }

  const userMsg = buildUserPrompt(tt, platform, content, ft, context);

  const first = await callOllama(
    ollama,
    route.primary,
    system,
    userMsg,
    route.timeoutMs
  );

  if (first.error) {
    if (route.fallback) {
      const fb = await callOllama(
        ollama,
        route.fallback,
        system,
        userMsg,
        route.timeoutMs
      );
      if (fb.error) {
        return errorOutput(["ollama_unavailable", first.error, fb.error]);
      }
      return toOutput(runGuard(fb.content, platform ?? "generic", ft, tt));
    }
    return errorOutput(["ollama_unavailable", first.error]);
  }

  const g1 = runGuard(first.content, platform ?? "generic", ft);

  if (g1.confidence < 0.7 && route.fallback) {
    const fb = await callOllama(
      ollama,
      route.fallback,
      system,
      userMsg,
      route.timeoutMs
    );
    if (!fb.error) {
      const g2 = runGuard(fb.content, platform ?? "generic", ft);
      if (g2.confidence > g1.confidence) {
        return toOutput(g2);
      }
    }
  }

  return toOutput(g1);
}

function toOutput(g: GuardResult): BadgrOutput {
  return {
    c: g.content,
    s: g.scores,
    v: g.violations,
    r: g.ready,
    conf: parseFloat(g.confidence.toFixed(3)),
    err: g.autoResolved,
    needs_claude_review: !g.ready || g.confidence < 0.7
  };
}

function errorOutput(errs: string[]): BadgrOutput {
  return {
    c: "",
    s: {},
    v: [],
    r: false,
    conf: 0,
    err: errs,
    needs_claude_review: true
  };
}
```

```ts
): { id: ToolId | null; confidence: number } {
  const text = naturalLanguage.toLowerCase();
  let best: { id: ToolId | null; confidence: number } = {
    id: null,
    confidence: 0
  };
  for (const cfg of TOOL_ALIASES) {
    for (const alias of cfg.aliases) {
      if (text.includes(alias)) {
        if (1 > best.confidence) best = { id: cfg.id, confidence: 1 };
      } else {
        const words = alias.split(/\s+/);
        const matched = words.filter(w => text.includes(w)).length;
        const score = matched / words.length;
        if (score > best.confidence && score >= 0.5) {
          best = { id: cfg.id, confidence: score };
        }
      }
    }
  }
  return best;
}

export function needsClarification(confidence: number): boolean {
  return confidence < 0.7;
}
```

## 7. RAG optimization (chunk size and doc types)

Your current RAG chunker splits on `##` headings and uses pre-defined chunks for the charts.[^2_5][^2_1]

```ts
function chunkByHeadings(
  text: string,
  source: string
): Array<{ text: string; tags: string[] }> {
  const parts = text.split(/^## /m);
  const chunks: Array<{ text: string; tags: string[] }> = [];
  const MAX_CHARS = 4500; // ~15% below a 5.5k-char "standard" chunk

  for (const part of parts) {
    const trimmed = part.trim();
    if (trimmed.length < 50) continue;
    const heading = trimmed.split("\n")[^2_0].trim();
    let remaining = trimmed;

    while (remaining.length > 0) {
      if (remaining.length <= MAX_CHARS) {
        chunks.push({
          text: `## ${remaining}`,
          tags: [source, heading.toLowerCase().replace(/\W+/g, "_").slice(0, 48)]
        });
        break;
      }
      // split at nearest paragraph boundary before MAX_CHARS
      const slice = remaining.slice(0, MAX_CHARS);
      const lastBreak =
        slice.lastIndexOf("\n\n") !== -1 ? slice.lastIndexOf("\n\n") : slice.length;
      const piece = remaining.slice(0, lastBreak).trim();
      chunks.push({
        text: `## ${piece}`,
        tags: [source, heading.toLowerCase().replace(/\W+/g, "_").slice(0, 48)]
      });
      remaining = remaining.slice(lastBreak).trim();
    }
  }

  return chunks;
}
```

This implements the “15% below standard norms” chunking requirement.[^2_5][^2_1]
